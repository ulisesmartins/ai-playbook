<script>
(function () {
  var BASE = 'https://github.com/dialpad/ai-playbook/raw/master/';

  function addDownloadLinks() {
    document.querySelectorAll('.dp-tree-wrap pre a').forEach(function (link) {
      if (link.classList.contains('dp-dl')) return;
      var text = link.textContent.trim();
      if (!text.endsWith('.md')) return;

      var href = link.getAttribute('href') || '';
      var clean = href.replace(/^(\.\.\/)+/, '').replace(/\/$/, '');

      // index.md links point to a directory — append /index.md
      // non-index links point to a MkDocs slug — append .md
      var path;
      if (text === 'index.md') {
        path = clean ? clean + '/index.md' : 'index.md';
      } else {
        path = clean + '.md';
      }

      var dl = document.createElement('a');
      dl.href = BASE + path;
      dl.className = 'dp-dl';
      dl.title = 'Download ' + text;
      dl.target = '_blank';
      dl.rel = 'noopener noreferrer';
      dl.setAttribute('aria-label', 'Download ' + text);
      dl.innerHTML = '<svg viewBox="0 0 16 16" width="11" height="11" fill="currentColor" aria-hidden="true"><path d="M7.47 10.78a.75.75 0 001.06 0l3.75-3.75a.75.75 0 00-1.06-1.06L8.75 8.44V1.75a.75.75 0 00-1.5 0v6.69L4.78 5.97a.75.75 0 00-1.06 1.06l3.75 3.75zM1.75 13.25a.75.75 0 000 1.5h12.5a.75.75 0 000-1.5H1.75z"/></svg>';

      link.insertAdjacentElement('afterend', dl);
    });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', addDownloadLinks);
  } else {
    addDownloadLinks();
  }
})();
